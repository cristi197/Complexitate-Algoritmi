cautare_binara.cpp 
int cautareBinara(int v[], int n, int x) {
  int st = 0, dr = n - 1;
  while (st <= dr) {
    int mij = (st + dr) / 2;
    if (v[mij] == x) return mij;
    if (v[mij] < x) st = mij + 1;
    else            dr = mij - 1;
  }
  return -1; // O(log n) ✓
}

// Complexitate: O(log n) timp · O(1) spațiu
// vs. căutare liniară: O(n) — de 1000x mai rapid
//    pentru n = 1.000.000 elemente
